 30                 /**

 32                  *
 33                  * @constructs
 34                  *
 35                  * @param {Object} params the default options to set on this Request.
 36                  * @param {String} [params.username] the user name to login to jango with
 37                  * @param {String} [params.password] password to login with
 38                  * @param {Boolean} [param.useSsl=true] set to false to not use SSL when connecting to jango.
 39                  * @param {String} [param.server="api.jangomail.com"] the jango server to connect to.
 40                  * @param {Integer} [param.port=443] the port to use when connecting to jango
 41                  * @param {Object} [param.contactInfo] Sets the list of target email address to send a message to and their field names.
 42                  *                                     The object should follow this JSON:

 54                 constructor : function(params) {
 55                     this.pattern = /SUCCESS\n(\d+)<\/string>$/;
 56                     this.postArray = {
 57                         Username : "",
 58                         Password : "",
 59                         ToGroups : "",
 60                         ToGroupFilter  : "",
 61                         ToOther : "",
 62                         ToWebDatabase : "",
 63                         Options : ""
 64                     };
 65                     comb.merge(this, params); // Merges all specified params into object
 66                 },
 67 
 68                 /**
 69                  * If set, this will tell Jango to hold off sending the mass email until
 70                  * the specified time.
 71                  *
 72                  * @param {Integer} stamp Epoch time in milliseconds.
 73                  */
 74                 setSendTime : function(stamp) {
 75                     var stampDate = new Date(stamp);
 76                     var adjusted = stampDate.getTime() + stampDate.getTimezoneOffset() * 60000; // Remove local timezone
 77                     var start = new Date(adjusted - this.__timeOffset); // Set it to EDT
 78                     var month = start.getMonth() + 1;
 79                     this.__time = month + "-" + start.getDate() + "-" + start.getFullYear() +
 80                             " " + start.toLocaleTimeString();
 81                 },
 82 
 83                 /**
 84                  * Clears the send time so any email sent will be immediate.
 85                  *
 86                  */
 87                 clearSendTime : function() {
 88                     this.__time = "";
 89                 },

 91 
 92                 __handleResponse : function(promise, serverResponse) {
 93                     var status = {
 94                         Status : serverResponse.statusCode,
 95                         Error :"",
 96                         JobID : 0
 97                     };
 98                     var data = "";
 99                     var jobIdCheck = this.pattern;
100                     serverResponse.setEncoding("UTF-8");
101                     serverResponse.on('data', function(chunk) {
102                         data += chunk.toString();
103                     });
104                     serverResponse.on("end", function() {
105                         if (status.Status != 200) {
106                             status.Error = data;
107                         } else {
108                             if (jobIdCheck.test(data)) {
109                                 status.JobID = jobIdCheck.exec(data)[1];
110                             } else {
111                                 status.JobID = -999;
112                             }
113                         }
114                         promise.callback(status);
115                     });
116                 },
117 
118                 /**
119                  * Makes the POST to Jango.
120                  *
121                  * @return {Promise} A promise that will receive a message like:
122                  * <pre class="code">
123                  *     { "code" : int,
124                  *       "error" : String,
125                  *       "jobid" : int
126                  *     }
127                  * <pre>
128                  */
129                 sendToJango : function() {
130                     var ret = new comb.Promise();
131                     this.postArray["Options"] = this.__toInformation;
132                     if (this.__time.length > 0) {
133                         this.postArray["Options"] += ",SendDate=" + this.__time;
134                     }
135                     var postBody = queryString.stringify(this.postArray);
136                     var options = {
137                         host: this.server,
138                         port: this.port,
139                         path: this.path,
140                         method: 'POST',
141                         headers: {
142                             "Content-Length": postBody.length,
143                             "Content-Type": "application/x-www-form-urlencoded"
144                         }
145                     };
146                     var client = this.ssl ? https : http;
147                     var req = client.request(options, comb.hitch(this, this.__handleResponse, ret));
148                     req.write(postBody);
149                     req.end();
150                     return ret;
151                 },

175                     /**
176                      * Sets the list of target email address to send a message to.  The array
177                      * should follow something like the following JSON notation:
178                      * <pre><code>
179                      *     [ {"firstName": "John", "lastName": "Smith", "email": "john.smith@home.com"}]
180                      * </code></pre>
181                      *
182                      * @field
183                      *
184                      * @throws Error if either the 'To' list is empty or one of the entries doesn't have an address.
185                      * @private
186                      */
187                     __emailList : function(toList) {
188                         var listLength = toList.length;
189                         if (listLength < 1) {
190                             throw "JangoRequestException:  \"To\" list was empty.";
191                         }
192                         var contacts = [], format = comb.string.format;
193                         for (var i = listLength - 1; i >= 0; i--) {
194                             contacts.push(format(this.__contactFormat, toList[i]));
195                         }
196                         this.postArray.ToOther = contacts.join("|");
197                     },

207                     },
208 
209                     /**
210                      * A list of field names that correspond to the email list.
211                      *
212                      * @field
213                      * @private
214                      */
215                     __fieldNames: function(fields) {
216                         this.__toInformation = "ToOtherRowDelimiter=|,ToOtherColDelimiter=c,ToOtherFieldNames=";
217                         var length = fields.length;
218                         this.__contactFormat = "";
219                         for (var i = 0; i < length; i++) {
220                             var field = fields[i];
221                             this.__toInformation += field + (i == (length - 1) ? "" : "|");
222                             this.__contactFormat += "{" + field + "}" + (i == (length - 1) ? "" : ",");
223                         }
224                         this.fields = fields;
225                     },
226

233                      *       "fieldNames" : ["firstName", "lastName", "emailAddress"]
234                      *     }
